1631. 最小体力消耗路径

并查集

public int minimumEffortPath(int[][] heights) {
        int rows = heights.length;
        int cols = heights[0].length;
        List<int[]> edges = new ArrayList<>();
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                int idx = i * cols + j;
                // 上方的节点构造边
                if (i > 0) {
                    edges.add(new int[]{idx, (i - 1) * cols + j, Math.abs(heights[i][j] - heights[i - 1][j])});
                }
                // 左边的节点构造边
                if (j > 0) {
                    edges.add(new int[]{idx, i * cols + j - 1, Math.abs(heights[i][j] - heights[i][j - 1])});
                }
            }
        }
        edges.sort(Comparator.comparingInt(o -> o[2]));
        int n = rows * cols;
        UnionFind uf = new UnionFind(n);
        for (int[] edge : edges) {
            uf.union(edge[0], edge[1]);
            if (uf.isConnected(0, n - 1)) {
                return edge[2];
            }
        }
        return 0;
    }

BFS

public int minimumEffortPath(int[][] heights) {
        int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
        int rows = heights.length;
        int cols = heights[0].length;
        boolean[][] visited = new boolean[rows][cols];
        // 到点 [i][j] 的最小权值
        int[][] dijkstra = new int[rows][cols];
        for (int[] row : dijkstra) {
            Arrays.fill(row, Integer.MAX_VALUE);
        }
        dijkstra[0][0] = 0;
        // 最小堆，每次都从权值最小的点开始找下一个点
        PriorityQueue<int[]> queue = new PriorityQueue<>(Comparator.comparingInt(o -> o[2]));
        queue.add(new int[]{0, 0, 0});
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            int x = cur[0];
            int y = cur[1];
            if (x == rows - 1 && y == cols - 1) {
                break;
            }
            if (visited[x][y]) {
                continue;
            }
            visited[x][y] = true;
            int dist = cur[2];
            for (int[] dir : dirs) {
                int nx = x + dir[0];
                int ny = y + dir[1];
                if (!inArea(nx, ny, rows, cols)) {
                    continue;
                }
                // 到达 [nx][ny] 的路径最大值
                int height = Math.max(dist, Math.abs(heights[x][y] - heights[nx][ny]));
                if (height < dijkstra[nx][ny]) {
                    dijkstra[nx][ny] = height;
                    queue.add(new int[]{nx, ny, height});
                }
            }
        }
        return dijkstra[rows - 1][cols - 1];
    }

    private boolean inArea(int x, int y, int rows, int cols) {
        return x >= 0 && x < rows && y >= 0 && y < cols;
    }
