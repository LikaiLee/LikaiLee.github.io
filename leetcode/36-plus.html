36 进制加法

每个字符转成十进制，相加时每 36 位进位。

public static void main(String[] args) {
        String num1 = "1b";
        String num2 = "2x";
        char[] s1 = num1.toCharArray();
        char[] s2 = num2.toCharArray();
        int carry = 0;
        StringBuilder res = new StringBuilder();
        int i = s1.length - 1, j = s2.length - 1;
        while (i >= 0 || j >= 0 || carry > 0) {
            int sum = carry;
            // num1 字符
            if (i >= 0) {
                sum += toDecimal(s1[i]);
                i--;
            }
            // num2 字符
            if (j >= 0) {
                sum += toDecimal(s2[j]);
                j--;
            }
            // 保留位
            int hold = sum % 36;
            res.insert(0, to36Char(hold));
            // 进位
            carry = sum / 36;
        }
        System.out.println(res.toString());
    }

    /**
     * 10 进制数字 => 36 进制字符
     *
     * @param num
     * @return
     */
    private static char to36Char(int num) {
        if (num < 10) {
            return (char) (num + '0');
        }
        return (char) ('a' + (num - 10));
    }

    /**
     * 36 进制字符 => 10 进制数字
     *
     * @param c
     * @return
     */
    private static int toDecimal(char c) {
        if (Character.isDigit(c)) {
            return c - '0';
        }
        return c - 'a' + 10;
    }
