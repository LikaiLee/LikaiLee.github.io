215. 数组中的第K个最大元素

快排

思想：排序后第 k 个最大元素位置 = len - k，每次 partition 后，都会将 pivot 移动到其最终位置，此时就可比较该位置是否是目标位置。

public int findKthLargest(int[] nums, int k) {
    int len = nums.length;
    // 排序后第 k 大的数所在位置
    int target = len - k;
    return quickSort(nums, target, 0, len - 1);
}

private int quickSort(int[] nums, int target, int left, int right) {
    int pivotPos = partition(nums, left, right);
    // 找到目标位置，返回
    if (pivotPos == target) {
        return nums[target];
    } else if (pivotPos > target) {
        // 比目标位置大，则往前查找
        return quickSort(nums, target, left, pivotPos - 1);
    } else {
        return quickSort(nums, target, pivotPos + 1, right);
    }
}

private int partition(int[] nums, int left, int right) {
    int pivot = nums[left];
    while (left < right) {
        while (left < right && nums[right] >= pivot) {
            right--;
        }
        nums[left] = nums[right];
        while (left < right && nums[left] <= pivot) {
            left++;
        }
        nums[right] = nums[left];
    }
    nums[left] = pivot;
    return left;
}

优先队列

小顶堆

public int findKthLargest(int[] nums, int k) {
    // 小顶堆 => 弹出比 目标小的 len - k 个元素，堆顶即为第 k 大
    PriorityQueue<Integer> minHeap = new PriorityQueue<>();
    for (int num : nums) {
        minHeap.add(num);
    }
    for (int i = 0; i < nums.length - k; i++) {
        minHeap.poll();
    }
    return minHeap.peek();
}

大顶堆

public int findKthLargest(int[] nums, int k) {
    int len = nums.length;
    // 大顶堆 => 弹出比 目标大的 k - 1 个元素，堆顶即为第 k 大
    PriorityQueue<Integer> maxHeap = new PriorityQueue<>(len, (a, b) -> b - a);
    for (int num : nums) {
        maxHeap.add(num);
    }
    for (int i = 0; i < k - 1; i++) {
        maxHeap.poll();
    }
    return maxHeap.peek();
}

堆排序

每次调整都把最大的元素放到后面，因此要找到第 K 大的元素只需要遍历 K 次。

或者：使用小顶堆

  1. 对 K 个元素先建堆
  2. 遍历剩下的元素，若大于堆顶，则加入堆并调整。
  3. 堆顶元素即为第 K 大的元素
public int findKthLargest(int[] nums, int k) {
    int len = nums.length;
    // 将数组整理成堆
    heapify(nums);

    // 循环不变量：区间 [0, i] 堆有序
    // 每次调整都把最大的元素放到后面
    // 因此要找到第 K 大的元素只需要遍历 K 次
    for (int i = len - 1; i > len - 1 - k; ) {
        // 把堆顶元素（当前最大）交换到数组末尾
        swap(nums, 0, i);
        // 逐步减少堆有序的部分
        i--;
        // 下标 0 位置下沉操作，使得区间 [0, i] 堆有序
        siftDown(nums, 0, i);
    }
    return nums[len - k];
}

/**
 * 将数组整理成堆（堆有序）
 *
 * @param nums
 */
private void heapify(int[] nums) {
    int len = nums.length;
    // 只需要从 i = (len - 1) / 2 这个位置开始逐层下移
    for (int i = (len - 1) / 2; i >= 0; i--) {
        siftDown(nums, i, len - 1);
    }
}

/**
 * @param nums
 * @param k    当前下沉元素的下标
 * @param end  [0, end] 是 nums 的有效部分
 */
private void siftDown(int[] nums, int k, int end) {
    while (2 * k < end) {
        int j = 2 * k + 1;
        // 左结点 < 右结点，j 指向右结点
        if (j < end && nums[j + 1] > nums[j]) {
            j++;
        }
        // 子结点大于父结点
        if (nums[j] > nums[k]) {
            swap(nums, j, k);
        } else {
            break;
        }
        k = j;
    }
}

private void swap(int[] nums, int i, int j) {
    int temp = nums[i];
    nums[i] = nums[j];
    nums[j] = temp;
}
