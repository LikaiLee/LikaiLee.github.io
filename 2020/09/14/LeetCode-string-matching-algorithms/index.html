public List<Integer> kmpSearch(String str, String pattern) {
    List<Integer> res = new ArrayList<>();
    // next 数组
    int[] next = new Solution().getNext(pattern);
    char[] s = str.toCharArray();
    char[] p = pattern.toCharArray();
    // i: 要匹配的字符串位置
    // j: 模式串位置
    int i = 0, j = 0;
    while (i < s.length && j < p.length) {
        // pattern 匹配成功，继续后移，查找是否有多个 pattern 在 str 中出现
        if (j == p.length - 1 && s[i] == p[j]) {
            res.add(i - j);
            j = next[j];
        }
        // 当前字符匹配成功，进行下一个字符的匹配
        if (j == -1 || s[i] == p[j]) {
            i++;
            j++;
        } else {
            // 匹配不成功，pattern 前移，从 str 失配处继续匹配
            j = next[j];
        }
    }
    return res;
}

/**
 * 获取模式串的 next 数组
 *
 * @param patternStr 模式串
 * @return next[]
 */
private int[] getNext(String patternStr) {
    char[] pattern = patternStr.toCharArray();
    int n = pattern.length;
    int[] next = new int[n];
    // 第 i 个位置前缀 = next[i + 1]
    next[0] = -1;
    // 初始前缀长度为 -1
    int len = -1;
    int i = 0;
    while (i < n - 1) {
        // 在上一个前缀的基础上新增的字符匹配，则 prefix[i] = len + 1
        if (len == -1 || pattern[i] == pattern[len]) {
            len++;
            // i + 1 位置存放 pattern[i] 的前缀长度
            i++;
            next[i] = len;
        } else {
            // 前缀不匹配，往前查找与 pattern[i] 相等的字符直到 len == -1
            len = next[len];
        }
    }
    return next;
}

BM

  1. 坏字符
    • 坏字符不在模式串中，移动模式串至坏字符后一位；
    • 坏字符在模式串中，移动模式串中最靠后的那个字符。
  2. 好后缀
    • 好后缀不在模式串中，移动模式串至好后缀的后一位；
    • 好后缀在模式串中，移动模式串至好后缀匹配的地方；
    • 好后缀的子串和模式串部分匹配，移动模式串至好后缀的子串重合的地方。
public int bmSearch(String s, String p) {
    char[] str = s.toCharArray();
    char[] pattern = p.toCharArray();
    int m = str.length;
    int n = pattern.length;
    // 生成坏字符在模式串中的位置数组
    int[] badChars = new int[256];
    badCharRule(pattern, badChars);

    // 长度 = i 的后缀子串在模式串中的起始位置
    int[] suffix = new int[n];
    // 长度 = i 的后缀子串是否有可匹配的前缀子串
    boolean[] hasPrefix = new boolean[n];
    goodSuffixRule(pattern, suffix, hasPrefix);

    int i = n - 1;
    while (i < m) {
        int j = n - 1;
        while (j >= 0 && pattern[j] == str[i]) {
            i--;
            j--;
            if (j == -1) {
                return i + 1;
            }
        }
        // 坏字符规则的移动次数
        int moveWithBC = j - badChars[str[i]];
        // 好后缀规则的移动次数
        int moveWithGS = Integer.MIN_VALUE;
        if (j < n - 1) {
            moveWithGS = moveWithGS(n, j, suffix, hasPrefix);
        }
        i += Math.max(moveWithBC, moveWithGS);
    }
    return -1;
}

/**
 * 好后缀规则下的移动次数
 *
 * @param n         模式串长度
 * @param j         失配位置
 * @param suffix    长度 = i 的后缀子串在模式串中的起始位置
 * @param hasPrefix 长度 = i 的后缀子串是否有可匹配的前缀子串
 * @return
 */
private int moveWithGS(int n, int j, int[] suffix, boolean[] hasPrefix) {
    int len = n - j - 1;
    // 后缀存在于模式串前缀中
    if (suffix[len] != -1) {
        return j - suffix[len] + 1;
    }
    // 若不存在前缀与后缀匹配，则找后缀的子串位置
    for (int i = len - 1; i >= 0; i--) {
        if (hasPrefix[i]) {
            return n - i;
        }
    }
    return n;
}

/**
 * 生成好后缀数组
 *
 * @param pattern
 * @param suffix
 * @param hasPrefix
 */
private void goodSuffixRule(char[] pattern, int[] suffix, boolean[] hasPrefix) {
    Arrays.fill(suffix, -1);
    int n = pattern.length;
    for (int i = 0; i < n - 1; i++) {
        int j = i;
        // 后缀子串的长度
        int len = 0;

        // 若字符匹配，则往前查找后缀
        while (j >= 0 && pattern[j] == pattern[n - 1 - len]) {
            j--;
            len++;
            suffix[len] = j + 1;
        }
        if (j == -1) {
            hasPrefix[len] = true;
        }
    }
}

/**
 * 记录 pattern 中每个字符的最靠后位置
 *
 * @param pattern
 * @param badChars
 */
private void badCharRule(char[] pattern, int[] badChars) {
    Arrays.fill(badChars, -1);
    for (int i = 0; i < pattern.length; i++) {
        badChars[pattern[i]] = i;
    }
}
