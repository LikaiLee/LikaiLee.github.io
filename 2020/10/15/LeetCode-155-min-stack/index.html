155. 最小栈

同步栈

class MinStack {

    LinkedList<Integer> stack;
    LinkedList<Integer> minStack;

    /**
     * initialize your data structure here.
     */
    public MinStack() {
        stack = new LinkedList<>();
        minStack = new LinkedList<>();
    }

    public void push(int x) {
        stack.push(x);
        if (!minStack.isEmpty()) {
            // 栈不为空，取最小值
            minStack.push(Math.min(x, getMin()));
        }
        // 栈空，x 就是最小的
        else {
            minStack.push(x);
        }
    }

    public void pop() {
        stack.pop();
        minStack.pop();
    }

    public int top() {
        return stack.peek();
    }

    public int getMin() {
        return minStack.peek();
    }
}

不同步栈

class MinStack {
    Deque<Integer> dataStack;
    Deque<Integer> minStack;

    /**
     * initialize your data structure here.
     */
    public MinStack() {
        dataStack = new LinkedList<>();
        minStack = new LinkedList<>();
    }

    public void push(int val) {
        dataStack.push(val);
        // 入栈元素 <= 最小值
        // [0, 1, 0] 需要重复最小值，否则 0 pop() 后栈为空
        if (minStack.isEmpty() || val <= getMin()) {
            minStack.push(val);
        }
    }

    public void pop() {
        int top = dataStack.pop();
        // 出栈的即为最小值
        if (!minStack.isEmpty() && top == getMin()) {
            minStack.pop();
        }
    }

    public int top() {
        return dataStack.peek();
    }

    public int getMin() {
        return minStack.peek();
    }
}
