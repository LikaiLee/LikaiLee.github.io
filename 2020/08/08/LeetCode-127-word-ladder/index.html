127. 单词接龙

和 433.最小基因那题很像，不过本题变异的字母是不确定的，不再是 ACGT，若用回溯会超时。

BFS

public int ladderLength(String beginWord, String endWord, List<String> wordList) {
        // ! endWord 不在 wordList 中直接返回
        if (!wordList.contains(endWord)) {
            return 0;
        }
        boolean[] used = new boolean[wordList.size()];
        LinkedList<String> queue = new LinkedList<>();
        // 将 beginWord 加入队列，标记访问数组
        queue.offer(beginWord);
        int start = wordList.indexOf(beginWord);
        if (start != -1) {
            used[start] = true;
        }
        int step = 0;
        while (!queue.isEmpty()) {
            step++;
            // 此时队列中的所有元素相当于上一个 word 可向下接龙的所有选择
            int size = queue.size();
            while (size-- > 0) {
                String word = queue.poll();
                for (int i = 0; i < wordList.size(); i++) {
                    String candidate = wordList.get(i);
                    // 若已经访问过 || word 无法接龙到 candidate，则跳过 
                    if (used[i] || !canLadder(word, candidate)) {
                        continue;
                    }
                    // candidate 到达 end，返回
                    if (endWord.equals(candidate)) {
                        return step + 1;
                    } else {
                        // 标记访问数组，将 candidate 加入队列
                        used[i] = true;
                        queue.offer(candidate);
                    }
                }
            }
        }
        // ！！！返回 0，不是 step
        return 0;
    }

    private boolean canLadder(String word, String candidate) {
        int count = 0;
        for (int i = 0; i < candidate.length(); i++) {
            if (word.charAt(i) != candidate.charAt(i)) {
                if (++count > 1) {
                    return false;
                }
            }
        }
        return true;
    }

双向 BFS

采用左右两个扩散集合，当集合重复即表示扩散结束。

public int ladderLength(String beginWord, String endWord, List<String> wordList) {
        // 不用 set 竟然超时了！
        Set<String> wordSet = new HashSet<>(wordList);
        if (!wordSet.contains(endWord)) {
            return 0;
        }
        // 标记已访问过的元素
        Set<String> used = new HashSet<>();
        // 左边扩散集合，每次都假设该集合元素最少，则扩散速度更快
        Set<String> beginSet = new HashSet<>();
        beginSet.add(beginWord);
        // 右边扩散集合
        Set<String> endSet = new HashSet<>();
        endSet.add(endWord);

        int step = 0;
        while (!beginSet.isEmpty() && !endSet.isEmpty()) {
            step++;
            // 使 beginSet 的元素最少
            if (beginSet.size() > endSet.size()) {
                Set<String> tmp = beginSet;
                beginSet = endSet;
                endSet = tmp;
            }
            // 下一层扩散的集合
            Set<String> nextLevel = new HashSet<>();
            for (String word : beginSet) {
                char[] words = word.toCharArray();
                for (int i = 0; i < words.length; i++) {
                    char old = words[i];
                    for (char c = 'a'; c <= 'z'; c++) {
                        if (old == c) {
                            continue;
                        }
                        words[i] = c;
                        String nextWord = String.valueOf(words);
                        // 新元素必须存在于提供的 list 中
                        if (wordSet.contains(nextWord)) {
                            // 右边集合包含该元素，说明扩散重复，直接返回
                            if (endSet.contains(nextWord)) {
                                return step + 1;
                            }
                            // 加入下一层扩散集合
                            if (!used.contains(nextWord)) {
                                nextLevel.add(nextWord);
                                // 标记已访问
                                used.add(nextWord);
                            }
                        }
                    }
                    words[i] = old;
                }
            }
            // 设置下一层的扩散集合
            beginSet = nextLevel;
        }

        return 0;
    }
