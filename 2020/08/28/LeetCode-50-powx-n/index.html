public double myPow(double x, int n) {
    long N = n;
    return N >= 0 ? getPow(x, N) : 1.0 / getPow(x, N);
}

private double getPow(double x, long n) {
    if (n == 0) {
        return 1.0;
    }
    double y = getPow(x, n / 2);
    return (n & 1) == 1 ? y * y * x : y * y;
}

public double myPow(double x, int n) {
    if (n == 0) {
        return 1;
    }
    long N = n;
    return N >= 0 ? getPow(x, N) : 1.0 / getPow(x, -N);
}

private double getPow(double x, long n) {
    double res = 1.0;
    while (n > 0) {
        // 最后一位为 1，表示此时 x^n 有效
        // 若为 0，增益为 0，结果不变
        if ((n & 1) == 1) {
            res *= x;
        }
        // 保存当前位的 x^i
        x *= x;
        // 删除最后一位，进行下一位计算
        n >>= 1;
    }
    return res;
}
