int rows;

/**

 */
boolean[][] visited;
/**
 * 移动方向
 */
int[][] dir = new int[][]{
        // 上: (x - 1, y)
        {-1, 0},
        // 右: (x, y + 1)
        {0, 1},
        // 下: (x + 1, y)
        {1, 0},
        // 左: (x, y - 1)
        {0, -1}
};

class Node {
    int x;
    int y;

    public Node(int x, int y) {
        this.x = x;
        this.y = y;
    }
}


public int numIslands(char[][] grid) {
    rows = grid.length;
    if (rows < 1) {
        return 0;
    }
    int count = 0;
    cols = grid[0].length;
    visited = new boolean[rows][cols];
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            // 当前位置未访问过 && 是陆地
            if (!visited[i][j] && grid[i][j] == '1') {
                // 只要未访问过且为 1 肯定存在一个岛屿
                // 需要做的就是找到这个岛屿连通的所有位置
                count++;

                Queue<Node> queue = new LinkedList<>();
                visited[i][j] = true;
                queue.offer(new Node(i, j));

                while (!queue.isEmpty()) {
                    Node cur = queue.poll();
                    for (int k = 0; k < dir.length; k++) {
                        int nextX = cur.x + dir[k][0];
                        int nextY = cur.y + dir[k][1];
                        // 新位置 未越界 && 未访问过 && 是陆地
                        if (inArea(nextX, nextY) && !visited[nextX][nextY] && grid[nextX][nextY] == '1') {
                            queue.offer(new Node(nextX, nextY));
                            // 及时标记，避免超时
                            visited[nextX][nextY] = true;
                        }
                    }
                }

            }
        }
    }
    return count;
}


private boolean inArea(int x, int y) {
    return x >= 0 && x < rows && y >= 0 && y < cols;
}

并查集

把所有的水连接到水域结点，若是陆地尝试与周围结点链接。

class Solution {
    int rows;
    int cols;
    int[][] dirs = {{1, 0}, {0, 1}};

    public int numIslands(char[][] grid) {
        rows = grid.length;
        if (rows == 0) {
            return 0;
        }
        cols = grid[0].length;
        if (cols == 0) {
            return 0;
        }
        int len = rows * cols;
        UnionFind uf = new UnionFind(len + 1);
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // 如果是 水，连接到 dummy 结点
                if (grid[i][j] == '0') {
                    uf.union(len, getIndex(i, j));
                } else {
                    // 如果是 陆地，尝试连接到周围的陆地
                    // 这里 dirs 只要 右和下两个方向就能遍历到周围的结点
                    // 因为在上一次遍历时已经访问过左和上方的结点
                    for (int[] dir : dirs) {
                        int newX = i + dir[0];
                        int newY = j + dir[1];
                        if (inArea(newX, newY) && grid[newX][newY] == '1') {
                            uf.union(getIndex(i, j), getIndex(newX, newY));
                        }
                    }
                }
            }
        }
        // 减去水域的结点才是所有的陆地结点
        return uf.getCount() - 1;
    }

    private int getIndex(int i, int j) {
        return i * cols + j;
    }

    private boolean inArea(int x, int y) {
        return x < rows && y < cols;
    }

}
