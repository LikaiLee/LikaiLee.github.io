212. 单词搜索 II
坑：查找前缀时，若找到 word，不能 return，否则递归结束会导致与 word 同前缀的单词丢失，如 aaba 和 aaab，只能找到 aaba。

还是对前缀树理解不够，找了好久才发现。

public class Solution {
    int[][] dirs = {{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    List<String> res = new ArrayList<>();
    boolean[][] used;
    int rows;
    int cols;

    public List<String> findWords(char[][] board, String[] words) {
        rows = board.length;
        if (rows == 0 || words.length == 0) {
            return res;
        }
        cols = board[0].length;
        used = new boolean[rows][cols];
        Trie trie = new Trie();
        for (String word : words) {
            trie.insert(word);
        }
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                backtrack(i, j, board, trie.root);
            }
        }
        return res;
    }

    private void backtrack(int i, int j, char[][] board, TrieNode node) {
        node = node.links[board[i][j] - 'a'];
        if (node == null) {
            return;
        }
        if (node.word != null) {
            if (!res.contains(node.word)) {
                res.add(node.word);
                // 一定不能 return，否则会丢失前缀
                // return;
            }
        }
        used[i][j] = true;
        for (int[] dir : dirs) {
            int newX = i + dir[0];
            int newY = j + dir[1];
            if (inArea(newX, newY) && !used[newX][newY]) {
                backtrack(newX, newY, board, node);
            }
        }
        used[i][j] = false;
    }

    private boolean inArea(int x, int y) {
        return x >= 0 && x < rows && y >= 0 && y < cols;
    }
}

class TrieNode {
    TrieNode[] links;
    String word;

    private final int R = 26;

    public TrieNode() {
        this.links = new TrieNode[R];
    }

    public boolean containsKey(char key) {
        return this.links[key - 'a'] != null;
    }

    public TrieNode get(char key) {
        return this.links[key - 'a'];
    }

    public void put(char key, TrieNode node) {
        this.links[key - 'a'] = node;
    }
}

class Trie {
    TrieNode root;

    public Trie() {
        root = new TrieNode();
    }

    public void insert(String word) {
        TrieNode node = root;
        for (int i = 0; i < word.length(); i++) {
            char key = word.charAt(i);
            if (!node.containsKey(key)) {
                node.put(key, new TrieNode());
            }
            node = node.get(key);
        }
        node.word = word;
    }
}
