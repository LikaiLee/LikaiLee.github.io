399. 除法求值

public double[] calcEquation(List<List<String>> equations, double[] values, List<List<String>> queries) {
    // 每个 equation 两个字母，映射到并查集 2 * size 个元素
    int size = equations.size();
    Map<String, Integer> map = new HashMap<>(size * 2);
    UnionFind uf = new UnionFind(size * 2);
    // equation 中的字母在并查集中的索引
    int id = 0;
    for (int i = 0; i < size; i++) {
        List<String> equation = equations.get(i);
        String x = equation.get(0);
        String y = equation.get(1);
        map.putIfAbsent(x, id++);
        map.putIfAbsent(y, id++);
        // x / y = values[i]
        uf.union(map.get(x), map.get(y), values[i]);
    }
    int querySize = queries.size();
    double[] res = new double[querySize];
    for (int i = 0; i < querySize; i++) {
        List<String> query = queries.get(i);
        String x = query.get(0);
        String y = query.get(1);
        if (!map.containsKey(x) || !map.containsKey(y)) {
            res[i] = -1.0;
        } else {
            double divide = uf.connected(map.get(x), map.get(y));
            res[i] = divide;
        }
    }
    return res;
}

private class UnionFind {
    private int[] parent;
    private double[] weight;

    public UnionFind(int size) {
        this.parent = new int[size];
        this.weight = new double[size];
        for (int i = 0; i < size; i++) {
            parent[i] = i;
            weight[i] = 1.0;
        }
    }

    public void union(int x, int y, double value) {
        int rootX = findRoot(x);
        int rootY = findRoot(y);
        if (rootX == rootY) {
            return;
        }
        // x 的根节点连到 y 的根节点
        // x 和 y 之间没有直接连接
        parent[rootX] = rootY;
        // x -> rootX -> rootY
        // x >(value) y -> rootY
        // 两条路径的权值相等
        weight[rootX] = value * weight[y] / weight[x];
    }

    private int findRoot(int x) {
        while (x != parent[x]) {
            int p = parent[x];
            // 路径压缩，所有节点都能连到根节点
            parent[x] = findRoot(parent[x]);
            // 更新权值
            // 从   x -> parent[x] -> p[p[x]]
            // 变为  x -> p[p[x]]
            weight[x] *= weight[p];
            x = parent[x];
        }
        return x;
    }

    public double connected(int x, int y) {
        int rootX = findRoot(x);
        int rootY = findRoot(y);
        // 根节点相同说明能转为同一个字母，才能相除
        return rootX == rootY ? weight[x] / weight[y] : -1.0;
    }
}
